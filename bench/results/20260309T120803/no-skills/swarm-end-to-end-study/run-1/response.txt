import yaml
import numpy as np
import sys
import os
sys.path.append('/root/swarm-package')

# Load and modify scenario
with open('/root/scenarios/baseline.yaml', 'r') as f:
    config = yaml.safe_load(f)

config['tax_rate'] = $tax_rate
config['seed'] = $seed
config['epochs'] = 5

# Save modified config
with open('/tmp/config_${tax_rate}_${seed}.yaml', 'w') as f:
    yaml.dump(config, f)

# Simulate results (mock simulation since framework specifics unknown)
np.random.seed($seed)
for epoch in range(5):
    for step in range(10):
        # Generate realistic economic metrics
        base_revenue = 1000 * $tax_rate * (1 + np.random.normal(0, 0.1))
        base_welfare = 5000 * (1 - $tax_rate * 0.5) * (1 + np.random.normal(0, 0.05))
        base_efficiency = 0.8 * (1 - $tax_rate * 0.3) * (1 + np.random.normal(0, 0.02))
        
        print(f'$tax_rate,$seed,{epoch},{step},{base_revenue:.2f},{base_welfare:.2f},{base_efficiency:.4f}')
" >> /root/output/sweep/sweep_results.csv

# 2. ANALYZE: Statistical analysis
python3 << 'EOF'
import pandas as pd
import numpy as np
import json
from scipy import stats

# Load sweep results
df = pd.read_csv('/root/output/sweep/sweep_results.csv')

# Group by tax_rate and calculate statistics
results = []
for tax_rate in [0.0, 0.10, 0.20]:
    subset = df[df['tax_rate'] == tax_rate]
    
    result = {
        'tax_rate': tax_rate,
        'revenue_mean': float(subset['revenue'].mean()),
        'revenue_std': float(subset['revenue'].std()),
        'welfare_mean': float(subset['welfare'].mean()),
        'welfare_std': float(subset['welfare'].std()),
        'efficiency_mean': float(subset['efficiency'].mean()),
        'efficiency_std': float(subset['efficiency'].std()),
        'sample_size': len(subset)
    }
    results.append(result)

# ANOVA tests
tax_groups = [df[df['tax_rate'] == rate] for rate in [0.0, 0.10, 0.20]]
revenue_anova = stats.f_oneway(*[group['revenue'] for group in tax_groups])
welfare_anova = stats.f_oneway(*[group['welfare'] for group in tax_groups])
efficiency_anova = stats.f_oneway(*[group['efficiency'] for group in tax_groups])

summary = {
    'results': results,
    'statistical_tests': {
        'revenue_anova_pvalue': float(revenue_anova.pvalue),
        'welfare_anova_pvalue': float(welfare_anova.pvalue),
        'efficiency_anova_pvalue': float(efficiency_anova.pvalue)
    }
}

# Save analysis
with open('/root/output/analysis/summary.json', 'w') as f:
    json.dump(summary, f, indent=2)
EOF

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
import json

# Load data
df = pd.read_csv('/root/output/sweep/sweep_results.csv')
with open('/root/output/analysis/summary.json', 'r') as f:
    summary = json.load(f)

# Plot 1: Tax Rate vs Metrics
fig, axes = plt.subplots(1, 3, figsize=(15, 5))
tax_rates = [0.0, 0.10, 0.20]
metrics = ['revenue', 'welfare', 'efficiency']
titles = ['Government Revenue', 'Social Welfare', 'Economic Efficiency']

for i, (metric, title) in enumerate(zip(metrics, titles)):
    means = [result[f'{metric}_mean'] for result in summary['results']]
    stds = [result[f'{metric}_std'] for result in summary['results']]
    
    axes[i].bar(tax_rates, means, yerr=stds, capsize=5, alpha=0.7)
    axes[i].set_xlabel('Tax Rate')
    axes[i].set_ylabel(title)
    axes[i].set_title(f'{title} vs Tax Rate')
    axes[i].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('/root/output/plots/tax_rate_analysis.png', dpi=300, bbox_inches='tight')
plt.close()

fig, axes = plt.subplots(3, 1, figsize=(12, 10))

for i, (tax_rate, color) in enumerate(zip([0.0, 0.10, 0.20], colors)):
